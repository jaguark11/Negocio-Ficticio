// main.js document.addEventListener("DOMContentLoaded", function() { const contactForm = document.getElementById("contactForm"); if (contactForm) { contactForm.addEventListener("submit", function(event) { event.preventDefault(); const email = document.getElementById("email"); const message = document.getElementById("message"); let valid = true; if (!email.value || !email.value.includes("@")) { alert("Por favor, ingresa un correo electrónico válido."); valid = false; } if (!message.value) { alert("Por favor, ingresa un mensaje."); valid = false; } if (valid) { alert("Gracias por ponerte en contacto con nosotros. Te responderemos pronto."); contactForm.reset(); } }); } });
